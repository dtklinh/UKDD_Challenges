## Check_maximal_repeat("ACGTAAATCGTAT", "CG")
Check_maximal_repeat <- function(S, t){
  t_idx <- gregexpr2(pattern = t, text = S)[[1]]
  if(length(t_idx)<=1){
    return(FALSE)
  }
  if(!1 %in% t_idx){
    ## expand to the left
    t_left_expand <- substr(S, t_idx[1]-1, t_idx[1]-1 + nchar(t))
    tmp_idx <- gregexpr2(t_left_expand, S)[[1]]
    tmp_idx <- tmp_idx +1
    if(length(setdiff(t_idx, tmp_idx))==0){
      return(FALSE)
    }
  }
  ## can only expand to the right
  t_right_expand <- substr(S, t_idx[1], t_idx[1] + nchar(t))
  tmp_idx <- gregexpr2(t_right_expand, S)[[1]]
  if(length(setdiff(t_idx, tmp_idx))==0){
    return(FALSE)
  }else{return(TRUE)}
}

Challenge 6

Observation

1) If a set X has N elements, its corresponding difference multiset \(\Delta(X)\) has N*(N-1)/2 elements.

2) Set X and set X + a (plus elementwise) have the same difference multiset.

3) Suppose X has N elements and in order \(X =\{x_1, x_2,...,x_N | x_1 < x_2 <...< x_N\}\), the biggest element in \(\Delta(X)\) has to be \(x_N - x_1\).

4) The following \(N-1\) elements could be used to determined the rest: \[\{(x_N - x_1), (x_N -x_2), ..., (x_N -x_{N-1})\}\]

Approach

1) Construct a linear system \(Ax=B\) to capture the differences.

\[\begin{aligned} x_N+0*x_{N-1}+...+0*x_2 -x_1 &= b_1 \\ x_N+0*x_{N-1}+...-x_2 +0*x_1 &= b_2 \\ ... \\ x_N - x_{N-1}+...+0*x_2 + 0*x_1 &= b_{N-1} \end{aligned}\]

where \(b_1\) is the biggest element in \(\Delta(X)\), and \(b_2 > b_3 > ... > b_{N-1}\).

2) The other differences are the difference among above linear equations. Thus the set $\{b_1, b_2,…,b_{N-1}\}$ must have the difference multiset as $\{b_N,b_{N+1},…\}$

3) \(X \rightarrow \Delta(X)\) is much easier to calculate than \(X \leftarrow \Delta(X)\)

## generate difference multiset of a given set
Diff_Multiset_generate <- function(X_input){
   X <- X_input[!duplicated(X_input)]
   if(length(X) != length(X_input)){
     print("coercing into a set")
   }
   N <- length(X)
   L <- c()
   for(i in 1:(N-1)){
     for (j in (i+1):N) {
       L <- c(L, abs(X[i]-X[j]))
     }
   }
   L <- sort(L)
   return(L)
}

## estimate original set from its corresponding difference multiset
Estimate_orignal <- function(L, max_iter_num = 100){
  ## L is multiset containing positive elements
  L <- sort(L, decreasing = T)
  N <- (1 + sqrt(8*length(L)+1))/2
  if(N != round(N)){
    stop("Number of elements in difference delta multiset is wrong!!!")
  }
  ## suppose that N >=3, in this case, when dealing with A*x = B, we will have the number of columns is greater than number of rows. Thus we muss apply decomposition to solve the matrix system
  ## A has dimension (N-1) x N
  A <- zeros(n=N-1, m = N)
  A[,N] <- 1
  A[,1:(N-1)] <- -1*diag(N-1)
  ## It is more tricky to construct B. We know that the first value of B has to be biggest value in L, but the rest are not sure, but they have to be distinct. 
  ## One way to tackle is that we compute all possibilites, and check the validation, as mentioned aboved.
  B <- c(L[1])
  L_reduced <- L[2:length(L)]
  L_reduced <- L_reduced[!duplicated(L_reduced)]
  All_cmbn <- combn(L_reduced, N-2)
  Solutions <- c()
  iter_num <- 0
  ##max_iter_num <- 100
  for(i in 1:ncol(All_cmbn)){
    if(iter_num > max_iter_num){
      ## the number of all combinations could be huge, thus only check with certain amount. 
      print(sprintf("Maximum number of iteration: %d has been reached! Increase if needed", max_iter_num))
      break}
    B <- c(L[1])
    B <- c(B, All_cmbn[,i]) %>% sort(decreasing = T)
    rest <- L[-match(B,L)] %>% sort(decreasing = T)
    B_delta <- Diff_Multiset_generate(B) %>% sort(decreasing = T)
    if(!all(rest == B_delta)){
      ## wrong guest, move to next combination
      iter_num <- iter_num +1
      next
    }
    asvd = svd(A)
    adiag = diag(1/asvd$d)
    #adiag <- adiag *1
    solution = asvd$v %*% adiag %*% t(asvd$u) %*% B
    ## shift solution
    min_val <- min(solution)
    solution <- solution + (1-min_val)
    Solutions <- cbind(Solutions, solution)
    ##print(Solutions)
    iter_num <- iter_num +1
    ##
  }
  return(Solutions)
}

Remarks

Overall, I have implemented those functions with expectations that users will give the proper inputs. The next step could be:

  • Better: Check and restrict users’ inputs and announce errors whenever encountering wrong inputs.

  • Best: Implementing unit testing for each functions.

The follows are the remarks for each challenge:

Challenge 1

At the moment the function only except input as a sequence of DNAString object, but it could be easily adjusted to accept string as character.

Start and end indices when sub-sequence a read could be thoroughly controlled in order to always deliver proper results.

Challenge 2

In this example, two reads are already aligned. In general case, we could apply alignment before calculate the transition/transversion ratio.

Challenge 3

It is important to use gregexpr2 from Biostrings package, the other similar function (base::gregexpr) will only return non-overlap substrings.

Challenge 4

  • The complexity of my function to find all maximal repeat of a string is at least $O(n^2)$ where $n$ is the length of the string. Thus, it can grow very quickly when the string is long.

  • I believe that there is more optimal algorithm tackling that problem, such as deplying suffix-tree algorithm when working with string.

  • I set the number threshold for maximal repeat to save time.

Challenge 5

I used Fastqc on Bioinformatics server and download the results to PC for processing.

Challenge 6

  • I also shared my observations and strategy to tackle this problem.

  • The complexity of this algirithm depends on the number of combinations from the $\Delta(X)$. Thus, it could be extremely big when dealing with a big multiset.

  • I believe that there is another way to limit the number of combinations, thus reducing the complexity.
